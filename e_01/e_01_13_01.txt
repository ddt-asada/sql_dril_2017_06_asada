アンケート回答テーブルから回答日時が新しい順に取得する。
作成日：2017年6月7日
作成者：浅田　知嗣

SELECT
name AS 氏名,
CASE
WHEN answer1 = 3 THEN 'ためになった'
WHEN answer1 = 2 THEN '普通'
WHEN answer1 = 1 THEN '役に立たない'
END AS 評価,
answer2 AS 感想

FROM
quest

ORDER BY
answered DESC
;
