社員テーブルから社員名、役職クラスを出力する。
作成日：2017年6月7日
作成者：浅田　知嗣

SELECT
CONCAT(l_name, f_name) AS 氏名,
CASE
WHEN class IN('部長', '課長') THEN '管理職'
WHEN class IN('主任', '担当') THEN '総合職'
ELSE '一般職'
END AS 役職クラス

FROM
employee
;
